cmake_minimum_required(VERSION 3.14)

project(AlienInvasion)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

option(BUILD_SHARED_LIBS "Build shared libraries" OFF)

include(FetchContent)
FetchContent_Declare(
    SFML
    GIT_REPOSITORY https://github.com/SFML/SFML.git
    GIT_TAG 2.6.x
    GIT_SHALLOW ON
    EXCLUDE_FROM_ALL
)

FetchContent_MakeAvailable(SFML)

add_executable(alien_invasion
    src/main.cpp
    src/AlienInvasion.cpp
    src/Ship.cpp
    src/Bullet.cpp
    src/Alien.cpp
    src/Settings.cpp
    src/GameStats.cpp
    src/PlayButton.cpp
    src/Scoreboard.cpp)
target_link_libraries(alien_invasion PRIVATE sfml-graphics)

target_compile_features(alien_invasion PRIVATE cxx_std_17)

if(WIN32)
    add_custom_command(
        TARGET alien_invasion
        COMMENT "Copy OpenAL DLL"
        PRE_BUILD COMMAND ${CMAKE_COMMAND} -E copy ${SFML_SOURCE_DIR}/extlibs/bin/$<IF:$<EQUAL:${CMAKE_SIZEOF_VOID_P},8>,x64,x86>/openal32.dll $<TARGET_FILE_DIR:alien_invasion>
        VERBATIM)
endif()
